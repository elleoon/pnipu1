#include <iostream>
using namespace std;

int main() {
	setlocale(LC_ALL, "Rus");

	int N;
	int s = 1;
	int j = 1;
	int i = 2;
	int max;
	int a;

	cout << "Введите количество элементов в послдовательности: ";
	cin >> N;
	while (N < 0) {
		cout << "Ошибка";
		cin >> N;

	}
	cout << "Введите первый элемент последовательности: ";
	cin >> a;
	max = a;

	while (i <= N) {
		cout << "Введите следующий элемент последовательности, с которым будем сравниваться предыдущий: ";
		cin >> a;
		if (a > max) {

			max = a;
			j = i;
			s = 1;
		}
		
		else
			if (a == max) {
				s = s + 1;
			}
		
		i++;

		
	}
	cout << "Максимальный элемент в последовательности: " << max << endl;
	cout << "Кол-во элментов совпадающих с максимальным: " << s << endl;
	cout << "Номер максимального элемента: " << j << endl;

	return 0;
